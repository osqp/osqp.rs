19macro_rules! u32_to_osqp_int {
20    ($name:ident, $value:expr) => {{
21        let value = $value;
22        assert!(
23            value as u64 <= ffi::osqp_int::max_value() as u64,
24            "{} must be smaller than the largest isize value",
25            stringify!($name)
26        );
27        value as ffi::osqp_int
28    }};
29}

31macro_rules! rust_type {
32    (float) => (float);
33    (u32) => (u32);
34    (option_u32) => (Option<u32>);
35    (bool) => (bool);
36    (linsys_solver) => (LinsysSolver);
37    (option_duration) => (Option<Duration>);
38}

40macro_rules! convert_rust_type {
41    ($name:ident, float, $value:expr) => ($value);
42    ($name:ident, u32, $value:expr) => (u32_to_osqp_int!($name, $value));
43    ($name:ident, option_u32, $value:expr) => (u32_to_osqp_int!($name, $value.unwrap_or(0)));
44    ($name:ident, bool, $value:expr) => ($value as ffi::osqp_int);
45    ($name:ident, linsys_solver, $value:expr) => (
46        match $value {
47            LinsysSolver::Unknown => ffi::OSQP_UNKNOWN_SOLVER,
48            LinsysSolver::Direct => ffi::OSQP_DIRECT_SOLVER,
49            LinsysSolver::Indirect => ffi::OSQP_INDIRECT_SOLVER,
50            LinsysSolver::__Nonexhaustive => unreachable!(),
51        }
52    );
53    ($name:ident, option_duration, $value:expr) => (
54        $value.map(|v| {
55            let mut secs = duration_to_secs(v);
56            // Setting time_limit to 0.0 disables the time limit to we treat a duration of zero as
57            // a very small time limit instead.
58            if secs == 0.0 {
59                secs = 1e-12;
60            }
61            secs
62        }).unwrap_or(0.0)
63    );
64}

66macro_rules! settings {
67    ($problem_ty:ty, $(
68        #[$doc:meta] $name:ident: $typ:ident $([$update_name:ident, $update_ffi:ident])*,
69    )*) => (
70        /// The settings used when initialising a solver.
71        pub struct Settings {
72            pub(crate) inner: ffi::OSQPSettings,
73        }
74
75        impl Settings {
76            $(
77                #[$doc]
78                pub fn $name(mut self, value: rust_type!($typ)) -> Settings {
79                    self.inner.$name = convert_rust_type!($name, $typ, value);
80                    Settings {
81                        inner: self.inner
82                    }
83                }
84            )*
85        }
86
87        impl Clone for Settings {
88            fn clone(&self) -> Settings {
89                unsafe {
90                    Settings {
91                        inner: ptr::read(&self.inner)
92                    }
93                }
94            }
95        }
96
97        impl Default for Settings {
98            fn default() -> Settings {
99                unsafe {
100                    let mut settings: ffi::OSQPSettings = mem::zeroed();
101                    ffi::osqp_set_default_settings(&mut settings);
102                    Settings {
103                        inner: settings
104                    }
105                }
106            }
107        }
108
109        unsafe impl Send for Settings {}
110        unsafe impl Sync for Settings {}
111
112        impl $problem_ty {
113            $($(
114                #[$doc]
115                pub fn $update_name(&mut self, value: rust_type!($typ)) {
116                    unsafe {
117                        let ret = ffi::$update_ffi(
118                            self.solver,
119                            convert_rust_type!($name, $typ, value)
120                        );
121                        if ret != 0 {
122                            panic!("updating {} failed", stringify!($name));
123                        }
124                    }
125                }
126            )*)*
127        }
128    );
129}

232fn duration_to_secs(dur: Duration) -> float {
233    dur.as_secs() as float + dur.subsec_nanos() as float * 1e-9
234}
